013  (ns clj-middleware.middleware
014    "Definition(s) of middleware used with the cooperation with Clojure Ring library.")
015  
016  (defn inject-configuration
017    "Inject configuration structure into the request parameter.
018       It means that the process request -> emender-waive -> response could be implemented
019       in a functional way."
020    [handler configuration]
021    (fn [request]
022      (handler (assoc request :configuration configuration))))
